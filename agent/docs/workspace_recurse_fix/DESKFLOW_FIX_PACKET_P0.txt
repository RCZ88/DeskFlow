WHY:
opencode is launching, but DeskFlow does not reliably mark it ready. The initialization prompt is then queued and later written raw, which breaks opencode's input behavior and produces visible "Agent initialization failed" UI even when there is no real console error. For bracketed-paste capable agents like opencode/claude, prompt sends must go through a safe paste+submit path, and readiness must not depend only on the old simple prompt regex.

CHANGE:

1. In src/main.ts, add these helpers near the AgentState / agentStates section.

```ts
function normalizeTerminalPasteText(data: string): string {
  return String(data ?? '').replace(/\r\n/g, '\n').replace(/\r/g, '\n');
}

function buildAgentInputPayload(data: string, agentType?: string): string {
  const normalized = normalizeTerminalPasteText(data);
  const cfg = getAgentConfig(agentType);

  // opencode / claude textareas handle bracketed paste correctly.
  // This prevents multi-line prompts from being interpreted as many separate Enter presses.
  if (cfg.bracketedPaste) {
    return `\x1b[200~${normalized}\x1b[201~\r`;
  }

  return `${normalized}\r`;
}

function getLastNonEmptyTerminalLine(buffer: string): string {
  const clean = stripAnsi(buffer);
  const lines = clean.split(/\r?\n/).map((l) => l.trim()).filter(Boolean);
  return lines[lines.length - 1] ?? '';
}

function flushPendingAgentWrites(id: string, st: AgentState) {
  if (!st.pendingWrites || st.pendingWrites.length === 0) return;

  const writes = [...st.pendingWrites];
  st.pendingWrites = [];

  for (const w of writes) {
    terminalManager.write(id, buildAgentInputPayload(w, st.agentType));
  }
}

function markAgentReady(id: string, st: AgentState) {
  if (st.phase !== 'launching') return;

  st.phase = 'ready';
  clearAgentTimeout(id);
  flushPendingAgentWrites(id, st);
  broadcast('agent:ready', { terminalId: id });
}

function hasEnoughAgentOutputToAcceptInput(st: AgentState): boolean {
  const cfg = getAgentConfig(st.agentType);
  if (!cfg.bracketedPaste) return false;

  const lastLine = getLastNonEmptyTerminalLine(st.dataBuffer);
  if (!lastLine) return false;
  if (looksLikeShell(lastLine)) return false;

  // opencode can render dynamic/TUI text that does not match "opencode >".
  // Once the process has emitted non-shell output, allow the paste path instead of timing out forever.
  return st.dataBuffer.trim().length > 0;
}
```
